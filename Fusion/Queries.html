Query to get Oracle Projects Basic Information


SELECT DISTINCT PPA.segment1 project_number ,PPA.name project_name ,HOU.name organization_name ,PPT.project_type ,TO_CHAR(PPA.start_date, 'DD-MON-YYYY') start_date ,TO_CHAR(PPA.completion_date, 'DD-MON-YYYY') completion_date ,(select xep.name from xle_entity_profiles xep where xep.legal_entity_id = PPA.LEGAL_ENTITY_ID) Legal_Entity ,(select PPA1.name from pjf_projects_all_vl PPA1 where PPA1.project_id = PPA.created_from_project_id) Template_name ,ppa.attribute1 Secondary_Service_Line ,ppa.attribute3 Opportunity_number ,ppa.project_id ,ppa.creation_date ,ppa.last_update_date FROM pjf_projects_all_vl PPA ,hr_all_organization_units HOU ,pjf_project_types_vl PPT WHERE PPA.carrying_out_organization_id = HOU.organization_id AND PPA.project_type_id = PPT.project_type_id
